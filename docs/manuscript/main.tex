\section{Methods}
\subsection{Assumptions}

Let $\{X(s): s\in D\}$ and $\{Y(s): s\in D\}$ be two spatial stochastic processes (random fields) representing two brain maps that are \textit{weakly stationary}, \textit{isotropic}, and \textit{strong mixing}. The processes are defined over a spatial domain $D$, which is either (1) a 3D brain volume or (2) a 2D manifold of the brain surface embedded in 3D space. In practice, data are observed at a finite set of discrete locations $S = \{s_1, s_2, \dots, s_n\}$ corresponding to (1) volumetric voxels or (2) surface vertices. For simplicity, assume $X(s)$ and $Y(s)$ are standardized to have zero mean and unit variance.\\

\textit{Weak stationary} implies a constant mean and variance (independent of spatial locations $s_1, s_2, \dots, s_n \in D$), and an auto-correlation function that only depends on spatial distance $h = s_i - s_j$:
\begin{align}
    \rho_X(h) = \text{corr}[X(s), X(s+h)] = \mathbb{E}[X(s) X(s+h)],
\end{align}

\noindent while \textit{isotropy} constrains the auto-correlation function to be the same in every direction:
\begin{align}
    \rho_X(h) = \rho_X(||h||).
\end{align}

\noindent \textit{Strong mixing} ($\alpha$-mixing) quantifies the decay of statistical dependence with spatial distance and ensures that observations from two spatial locations become asymptotically independent as the separation between them increases.\\


\subsection{Hypothesis Testing with Subsampling}

For analysis, we are interested in the cross-correlation at the same spatial locations (colocated correlation):
\begin{align}
    \rho_{XY} = \text{corr}[X(s), Y(s)] = \mathbb{E}[X(s) Y(s)],
\end{align}

\noindent and the corresponding hypothesis test:
\begin{align}
    H_0: \rho_{XY} = 0 \quad\text{against}\quad H_1:\rho_{XY} \ne 0.
\end{align}

The subsampling methods is a non-parametric approach to test the null hypothesis, $H_0: \rho_{XY} = 0$, by empirically constructing the sampling distribution of the test statistic under the null. This procedure is valid under the stated assumptions of \textit{weak stationarity}, \textit{isotropy}, and \textit{strong mixing}, without requiring a parametric form of the sampling distribution.\\

The subsampling procedure is as follows:

\begin{enumerate}
    \item Estimate the cross-correlation from the full sample of $n$ locations:
    \begin{align}
        \hat\rho_{XY} = \frac{1}{n} \sum_{i=1}^n X(s_i) Y(s_i)
    \end{align}

    \item Subsample $B$ overlapping spatial patches and re-estimate the cross-correlation. Each patch contains $m$ neighboring voxels/vertices, where $m$ is chosen such that $m\to\infty$ and $m/n \to 0$ as $n\to\infty$. For each subsample $j=1, 2, \dots, B$, the sample cross-correlation $\hat\rho^*_{XY}$ is re-estimated using the $m$ locations within the patch. 

    \item Calculate the percentile interval from the quantiles of the empirical approximation of the sampling distribution, $\{\hat\rho^*_{XY, 1}, \hat\rho^*_{XY, 2}, \dots, \hat\rho^*_{XY, B}\}$.
\end{enumerate}

\subsection*{Data and Parameters}

Data generating process/Population distribution
\begin{align}
    P
\end{align}

Stochastic process
\begin{align}
    \{X_t: t\in\mathbb{Z}\} = \{\dots, X_{-1}, X_0, X_1, \dots\}
\end{align}

Observed sample from $P\in\mathcal{P}$
\begin{align}
    X_n = \{X_1, \dots, X_n\}
\end{align}

Empirical distribution of the sample
\begin{align}
    \hat{P}_n
\end{align}

Parameter of interest
\begin{align}
    \theta = \theta(P)
\end{align}

Estimator of $\theta(P)$
\begin{align}
    \hat{\theta}_n = \hat{\theta}_n(X_1, \dots, X_n)
\end{align}

\subsection*{Sampling Distributions}

Normalizing constant/Rate of convergence (e.g. $\sqrt{n}$)
\begin{align}
    \tau_n
\end{align}

Root, a function whose sampling distribution is of interest. It can also be studentized.
\begin{align}
    R_n = \tau_n(\hat\theta_n - \theta)
\end{align}

Finite sampling distribution of the root $R_n$ under $P$
\begin{align}
    J_n(P)
\end{align}

Asymptotic sampling distribution, $J_n(P)$ as $n\to\infty$
\begin{align}
    J(P)
\end{align}


\subsection*{Bootstraping}
Bootstrapping approximates the sampling distribution $J_n(P)$ by simulating from an estimated population, $\hat{P}_n$, based on the plug-in principle.\\

Bootstrap sample from $\hat{P}_n$, \textit{iid} with replacement
\begin{align}
    X^*_n = \{X^*_1,\dots,X^*_n\}
\end{align}

Bootstrap estimator
\begin{align}
    \hat\theta^*_n = \hat\theta_n(X^*_1,\dots,X^*_n)
\end{align}

Bootstrap root, centered around $\hat\theta_n$ which is the \textit{theoretical} parameter value in the bootstrap world ($\theta(\hat{P}_n) \approx \hat{\theta}_n$).
\begin{align}
    R^*_n = \tau_n(\hat\theta^*_n - \hat\theta_n)
\end{align}

Bootstrap distribution estimator. The theoretical distribution of $R_n^*$ under resampling from $\hat{P}_n$, which approximates the true sampling distribution $J_n(P)$. 
\begin{align}
    J_n(\hat{P}_n)
\end{align}

The notation $J_n(\hat P_n)$ represents this theoretical distribution, which is then estimated using Monte Carlo simulation (bootstrap replicates).

\subsection*{Subsampling}

Subsampling approximates the sampling distribution $J_n(P)$ by recomputing the statistic on smaller subsamples $X_b$ of the observed sample $X_n$. This method assumes that the sampling distribution at size $b$, $J_b(P)$, is a good approximation of the sampling distribution at size $n$, $J_n(P)$.\\

Subsample from $P$ without replacement. The $i^{th}$ subsample of size $b$, for $i=1, \dots, N_b$, where $N_b = n-b+1$ is the number of subsamples.
\begin{align}
    X_{b,i} = \{X_i, \dots, X_{i+b-1}\}
\end{align}

Subsample estimator
\begin{align}
    \hat\theta_{b,i} = \hat\theta_b(X_{b,i})
\end{align}

Subsample distribution estimators, which are the empirical distribution functions of the recomputed statistics.\\

For confidence intervals, $L_{n,b}(x)$ is the empirical distribution of the centered subsample statistics
\begin{align}
L_{n,b}(x) = \frac{1}{N_b} \sum_{i=1}^{N_b} \mathbf{1}{\tau_b(\hat{\theta}_{b,i} - \hat{\theta}_n) \le x}
\end{align}

For hypothesis testing, $G_{n,b}(x)$ is the empirical distribution of the subsample statistics centered at the null value $\theta_0$
\begin{align}
G_{n,b}(x) = \frac{1}{N_b} \sum_{i=1}^{N_b} \mathbf{1}{\tau_b(\hat{\theta}_{b,i} - \theta_0) \le x}
\end{align}

This notation emphasizes that they are direct empirical estimators of $J_b(P)$, not theoretical distributions derived from a plug-in model.
